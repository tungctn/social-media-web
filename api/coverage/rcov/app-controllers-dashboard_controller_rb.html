235   def generate_months_range(start_date, end_date)
 
236     start_month = start_date.beginning_of_month
 
237     end_month = end_date.beginning_of_month
 
238 
239     months_range = [start_month.month]
 
240     current_month = start_month
 
241 
242     while current_month < end_month

244       months_range << current_month.month
